滑块验证
实例化xh_slider；
el: id名;
success: 验证成功返回参数“success”
code
  1. <script type="text/javascript">
  2.     var a = new xh_slider({
  3.         el: '#slider',
  4.         success: function(data) {
  5.             if (data == 'success') {
  6.                 // 验证成功之后的操作
  7.                 $(this).xh_prompt('success', '验证成功', 2000);
  8.             }
  9.         }
  10.     });
  11. </script>
